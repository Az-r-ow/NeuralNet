124 void init(int numRows)
125 {
126 // First and foremost init the biases and the outputs
127 double mean = 0, stddev = 0;
128 this->weights = Eigen::MatrixXd::Zero(numRows, nNeurons);
129
130 // This is going to be used for testing
131 if (this->weightInit == WEIGHT_INIT::CONSTANT)
132 {
133 this->weights = Eigen::MatrixXd::Constant(numRows, nNeurons, 1);
134 return;
135 }
136
137 // calculate mean and stddev based on init algo
138 switch (this->weightInit)
139 {
140 case WEIGHT_INIT::GLOROT:
141 // sqrt(fan_avg)
142 stddev = sqrt(static_cast<double>(2) / (numRows + nNeurons));
143 break;
144 case WEIGHT_INIT::HE:
145 // sqrt(2/fan_in)
146 stddev = sqrt(2.0 / numRows);
147 break;
148 case WEIGHT_INIT::LECUN:
149 // sqrt(1/fan_in)
150 stddev = sqrt(1.0 / numRows);
151 break;
152 default:
153 break;
154 }
155
156 // Init the weights
157 this->weightInit == WEIGHT_INIT::RANDOM ? randomWeightInit(&(this->weights), -1, 1) : randomDistMatrixInit(&(this->weights), mean, stddev);
158 }

214 Eigen::MatrixXd computeOutputs(Eigen::MatrixXd inputs)
215 {
216 // Initialize the biases based on the input's size
217 if (biases.rows() == 0 && biases.cols() == 0)
218 {
219 biases = Eigen::MatrixXd::Constant(1, nNeurons, bias);
220 }
221
222 // Weighted sum
223 Eigen::MatrixXd wSum = inputs * weights;
224
225 wSum.rowwise() += biases.row(0);
226
227 outputs = activate(wSum);
228 return outputs;
229 };
