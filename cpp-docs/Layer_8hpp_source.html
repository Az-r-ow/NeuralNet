105 void init(int numRows) {
106 // First and foremost init the biases and the outputs
107 double mean = 0, stddev = 0;
108 this->weights = Eigen::MatrixXd::Zero(numRows, nNeurons);
109
110 // This is going to be used for testing
111 if (this->weightInit == WEIGHT_INIT::CONSTANT) {
112 this->weights = Eigen::MatrixXd::Constant(numRows, nNeurons, 1);
113 return;
114 }
115
116 // calculate mean and stddev based on init algo
117 switch (this->weightInit) {
118 case WEIGHT_INIT::GLOROT:
119 // sqrt(fan_avg)
120 stddev = sqrt(static_cast<double>(2) / (numRows + nNeurons));
121 break;
122 case WEIGHT_INIT::HE:
123 // sqrt(2/fan_in)
124 stddev = sqrt(2.0 / numRows);
125 break;
126 case WEIGHT_INIT::LECUN:
127 // sqrt(1/fan_in)
128 stddev = sqrt(1.0 / numRows);
129 break;
130 default:
131 break;
132 }
133
134 // Init the weights
135 this->weightInit == WEIGHT_INIT::RANDOM
136 ? randomWeightInit(&(this->weights), -1, 1)
137 : randomDistMatrixInit(&(this->weights), mean, stddev);
138 }

195 Eigen::MatrixXd computeOutputs(Eigen::MatrixXd inputs) {
196 // Initialize the biases based on the input's size
197 if (biases.rows() == 0 && biases.cols() == 0) {
198 biases = Eigen::MatrixXd::Constant(1, nNeurons, bias);
199 }
200
201 // Weighted sum
202 Eigen::MatrixXd wSum = inputs * weights;
203
204 wSum.rowwise() += biases.row(0);
205
206 outputs = activate(wSum);
207 return outputs;
208 };
