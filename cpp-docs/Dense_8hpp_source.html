116 void init(int numRows) override {
117 // First and foremost init the biases and the outputs
118 double mean = 0, stddev = 0;
119 this->weights = Eigen::MatrixXd::Zero(numRows, nNeurons);
120
121 // This is going to be used for testing
122 if (this->weightInit == WEIGHT_INIT::CONSTANT) {
123 this->weights = Eigen::MatrixXd::Constant(numRows, nNeurons, 1);
124 return;
125 }
126
127 // calculate mean and stddev based on init algo
128 switch (this->weightInit) {
129 case WEIGHT_INIT::GLOROT:
130 // sqrt(fan_avg)
131 stddev = sqrt(static_cast<double>(2) / (numRows + nNeurons));
132 break;
133 case WEIGHT_INIT::HE:
134 // sqrt(2/fan_in)
135 stddev = sqrt(2.0 / numRows);
136 break;
137 case WEIGHT_INIT::LECUN:
138 // sqrt(1/fan_in)
139 stddev = sqrt(1.0 / numRows);
140 break;
141 default:
142 break;
143 }
144
145 // Init the weights
146 this->weightInit == WEIGHT_INIT::RANDOM
147 ? randomWeightInit(&(this->weights), -1, 1)
148 : randomDistMatrixInit(&(this->weights), mean, stddev);
149 }

160 Eigen::MatrixXd computeOutputs(Eigen::MatrixXd inputs) override {
161 // Initialize the biases based on the input's size

167 Eigen::MatrixXd wSum = inputs * weights;

169 wSum.rowwise() += biases.row(0);
