108 void init(int numRows) override {
109 // First and foremost init the biases and the outputs
110 double mean = 0, stddev = 0;
111 this->weights = Eigen::MatrixXd::Zero(numRows, nNeurons);
112
113 // This is going to be used for testing
114 if (this->weightInit == WEIGHT_INIT::CONSTANT) {
115 this->weights = Eigen::MatrixXd::Constant(numRows, nNeurons, 1);
116 return;
117 }
118
119 // calculate mean and stddev based on init algo
120 switch (this->weightInit) {
121 case WEIGHT_INIT::GLOROT:
122 // sqrt(fan_avg)
123 stddev = sqrt(static_cast<double>(2) / (numRows + nNeurons));
124 break;
125 case WEIGHT_INIT::HE:
126 // sqrt(2/fan_in)
127 stddev = sqrt(2.0 / numRows);
128 break;
129 case WEIGHT_INIT::LECUN:
130 // sqrt(1/fan_in)
131 stddev = sqrt(1.0 / numRows);
132 break;
133 default:
134 break;
135 }
136
137 // Init the weights
138 this->weightInit == WEIGHT_INIT::RANDOM
139 ? randomWeightInit(&(this->weights), -1, 1)
140 : randomDistMatrixInit(&(this->weights), mean, stddev);
141 }

152 Eigen::MatrixXd computeOutputs(Eigen::MatrixXd inputs) override {
