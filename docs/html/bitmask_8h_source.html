45#define BITMASK_W unsigned long int

86/* Returns nonzero if the bit at (x,y) is set. Coordinates start at
87 (0,0) */
88static INLINE int
89bitmask_getbit(const bitmask_t *m, int x, int y)
90{
91 return (m->bits[x / BITMASK_W_LEN * m->h + y] &
92 BITMASK_N(x & BITMASK_W_MASK)) != 0;
93}
94
95/* Sets the bit at (x,y) */
96static INLINE void
97bitmask_setbit(bitmask_t *m, int x, int y)
98{
99 m->bits[x / BITMASK_W_LEN * m->h + y] |= BITMASK_N(x & BITMASK_W_MASK);
100}
101
102/* Clears the bit at (x,y) */
103static INLINE void
104bitmask_clearbit(bitmask_t *m, int x, int y)
105{
106 m->bits[x / BITMASK_W_LEN * m->h + y] &= ~BITMASK_N(x & BITMASK_W_MASK);
107}
