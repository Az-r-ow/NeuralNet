886static PG_INLINE PyObject *
887pg_tuple_couple_from_values_int(int val1, int val2)
888{
889 /* This function turns two input integers into a python tuple object.
890 * Currently, 5th November 2022, this is faster than using Py_BuildValue
891 * to do the same thing.
892 */
893 PyObject *tup = PyTuple_New(2);
894 if (!tup) {
895 return NULL;
896 }
897
898 PyObject *tmp = PyLong_FromLong(val1);
899 if (!tmp) {
900 Py_DECREF(tup);
901 return NULL;
902 }
903 PyTuple_SET_ITEM(tup, 0, tmp);
904
905 tmp = PyLong_FromLong(val2);
906 if (!tmp) {
907 Py_DECREF(tup);
908 return NULL;
909 }
910 PyTuple_SET_ITEM(tup, 1, tmp);
911
912 return tup;
913}
