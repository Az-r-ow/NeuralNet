45 {
46 assert(param.rows() == gradients.rows() && param.cols() == gradients.cols());
47
48 // increment time step
49 t = t + 1;
50
51 if (m.rows() == 0 || m.cols() == 0)

53 // Initialize moment matrices m and v
54 m = Eigen::MatrixBase<Derived1>::Zero(param.rows(), param.cols());
55 v = Eigen::MatrixBase<Derived1>::Zero(param.rows(), param.cols());
56 }
57
58 // update biased first moment estimate
59 m = (beta1 * m).array() + ((1 - beta2) * gradients.array()).array();
60
61 // updated biased second raw moment estimate
62 v = (beta2 * v).array() + ((1 - beta2) * (gradients.array() * gradients.array())).array();
63

66
67 // compute bias-corrected second raw moment estimate
68 double beta2_t = std::pow(beta2, t);
69
70 double alpha_t = alpha * (sqrt(1 - beta2_t) / (1 - beta1_t));
71
72 // update param
73 param = param.array() - alpha_t * (m.array() / (v.array().sqrt() + epsilon));
74 }
